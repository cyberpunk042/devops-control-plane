/* Integrations Tab JS — Artifacts Card loadArtifactsCard(), target display, build actions. Depends on: _integrations_init.html (card registry) */ // ── Artifacts Card ────────────────────────────────────────────── async function loadArtifactsCard() { const badge = document.getElementById('int-artifacts-badge'); const detail = document.getElementById('int-artifacts-detail'); try { const cached = cardCached('artifacts'); let data; if (cached) { data = cached; } else { data = await api('/artifacts/targets'); cardStore('artifacts', data); } const targets = data.targets || []; if (targets.length === 0) { badge.className = 'status-badge degraded'; badge.innerHTML = 'No targets'; detail.innerHTML = `

No artifact targets configured yet.

`; return; } // Count built targets const built = targets.filter(t => t.build).length; const total = targets.length; badge.className = built === total && built > 0 ? 'status-badge ok' : 'status-badge degraded'; badge.innerHTML = `${built}/${total} built`; // ── Target rows ── const listHtml = targets.map(t => { const buildInfo = t.build; const buildOk = !!buildInfo; const buildAgo = buildInfo ? _timeAgo(buildInfo.built_at) : ''; const duration = buildInfo ? `${buildInfo.duration_ms}ms` : ''; const timeInfo = buildAgo ? `${buildAgo}${duration ? ' · ' + duration : ''}` : ''; // Kind badge const kindColors = { local: 'var(--success)', package: '#818cf8', bundle: 'var(--warning)', script: 'var(--text-muted)', container: '#06b6d4', }; const kindColor = kindColors[t.kind] || 'var(--text-muted)'; // Status dots: configured → built const dotsHtml = ` ● → ●`; return `
${dotsHtml} ${esc(t.name)} ${esc(t.kind)} ${esc(t.builder)} ${timeInfo}
`; }).join(''); // ── Pipeline actions ── const pipelineHtml = `
Pipeline: Targets → Build → Distribute
`; // ── Builders row ── const builderTypes = [...new Set(targets.map(t => t.builder))]; const builderChips = builderTypes.map(b => `✓ ${esc(b)}` ).join(' '); const buildersHtml = builderChips ? `
Builders: ${builderChips}
` : ''; detail.innerHTML = `
${listHtml} ${pipelineHtml} ${buildersHtml}
`; } catch (e) { badge.className = 'status-badge failed'; badge.innerHTML = 'Error'; detail.innerHTML = `

${esc(e.message)}

`; } } // ── Artifact actions ── async function removeArtifactTarget(name) { if (!confirm(`Remove artifact target "${name}"?`)) return; try { await api(`/artifacts/targets/${name}`, { method: 'DELETE' }); toast(`Removed target: ${name}`, 'success'); cardInvalidate('artifacts'); loadArtifactsCard(); } catch (e) { toast(`Error: ${e.message}`, 'error'); } } function addArtifactTargetModal() { toast('Add artifact target — coming soon', 'info'); } function buildArtifactTarget(name) { // Show build log panel const card = document.getElementById('int-artifacts-card'); let logPanel = document.getElementById('artifact-build-log'); if (!logPanel) { logPanel = document.createElement('div'); logPanel.id = 'artifact-build-log'; logPanel.style.cssText = 'margin-top:var(--space-md);background:var(--bg-inset);border:1px solid var(--border-subtle);border-radius:var(--radius-sm);padding:0.5rem;max-height:300px;overflow-y:auto;font-family:var(--font-mono);font-size:0.7rem;line-height:1.5;color:var(--text-secondary)'; card.appendChild(logPanel); } logPanel.innerHTML = `
⏳ Building ${esc(name)}…
`; logPanel.style.display = 'block'; // Disable build buttons during build const btns = card.querySelectorAll('button[onclick*="buildArtifact"]'); btns.forEach(b => { b.disabled = true; }); // SSE stream fetch(`/api/artifacts/build/${encodeURIComponent(name)}/stream`, { method: 'POST' }) .then(response => { const reader = response.body.getReader(); const decoder = new TextDecoder(); let buffer = ''; function read() { reader.read().then(({ done, value }) => { if (done) { _artifactBuildDone(name, card, btns); return; } buffer += decoder.decode(value, { stream: true }); const lines = buffer.split('\n'); buffer = lines.pop(); // Keep incomplete line in buffer for (const raw of lines) { if (!raw.startsWith('data: ')) continue; const line = raw.slice(6); if (line === '__done__') { _artifactBuildDone(name, card, btns); return; } if (line.startsWith('__BUILD_RESULT__:')) { // Parse result but don't display as log continue; } const div = document.createElement('div'); // Color success/error lines if (line.startsWith('✅')) div.style.color = 'var(--success)'; else if (line.startsWith('❌')) div.style.color = 'var(--error)'; else if (line.startsWith('━━━')) { div.style.color = 'var(--accent)'; div.style.fontWeight = '600'; } div.textContent = line; logPanel.appendChild(div); logPanel.scrollTop = logPanel.scrollHeight; } read(); }); } read(); }) .catch(err => { logPanel.innerHTML += `
❌ Stream error: ${esc(err.message)}
`; btns.forEach(b => { b.disabled = false; }); }); } function _artifactBuildDone(name, card, btns) { btns.forEach(b => { b.disabled = false; }); cardInvalidate('artifacts'); loadArtifactsCard(); toast(`Build complete: ${name}`, 'success'); } async function buildAllArtifacts() { try { const data = await api('/artifacts/targets'); const targets = data.targets || []; for (const t of targets) { buildArtifactTarget(t.name); // Wait a bit between builds to avoid overlap await new Promise(r => setTimeout(r, 500)); } } catch (e) { toast(`Error: ${e.message}`, 'error'); } }
