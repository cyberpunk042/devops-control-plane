/* Integrations Tab JS — Artifacts Card loadArtifactsCard(), target display, build actions. Depends on: _integrations_init.html (card registry) */ // ── Artifacts Card ────────────────────────────────────────────── async function loadArtifactsCard() { const badge = document.getElementById('int-artifacts-badge'); const detail = document.getElementById('int-artifacts-detail'); try { const cached = cardCached('artifacts'); let data; if (cached) { data = cached; } else { data = await api('/artifacts/targets'); cardStore('artifacts', data); } const targets = data.targets || []; if (targets.length === 0) { badge.className = 'status-badge degraded'; badge.innerHTML = 'No targets'; detail.innerHTML = `

No artifact targets configured yet.

`; return; } // Count built targets const built = targets.filter(t => t.build).length; const total = targets.length; badge.className = built === total && built > 0 ? 'status-badge ok' : 'status-badge degraded'; badge.innerHTML = `${built}/${total} built`; // ── Target rows ── const listHtml = targets.map(t => { const buildInfo = t.build; const buildOk = !!buildInfo; const buildAgo = buildInfo ? _timeAgo(buildInfo.built_at) : ''; const duration = buildInfo ? `${buildInfo.duration_ms}ms` : ''; const timeInfo = buildAgo ? `${buildAgo}${duration ? ' · ' + duration : ''}` : ''; // Kind badge const kindColors = { local: 'var(--success)', package: '#818cf8', bundle: 'var(--warning)', script: 'var(--text-muted)', container: '#06b6d4', }; const kindColor = kindColors[t.kind] || 'var(--text-muted)'; // Status dots: configured → built const dotsHtml = ` ● → ●`; return `
${dotsHtml} ${esc(t.name)} ${esc(t.kind)} ${esc(t.builder)} ${timeInfo}
`; }).join(''); // ── Pipeline actions ── const pipelineHtml = `
Pipeline: Targets → Build → Distribute
`; // ── Builders row ── const builderTypes = [...new Set(targets.map(t => t.builder))]; const builderChips = builderTypes.map(b => `✓ ${esc(b)}` ).join(' '); const buildersHtml = builderChips ? `
Builders: ${builderChips}
` : ''; detail.innerHTML = `
${listHtml} ${pipelineHtml} ${buildersHtml}
`; } catch (e) { badge.className = 'status-badge failed'; badge.innerHTML = 'Error'; detail.innerHTML = `

${esc(e.message)}

`; } } // ── Artifact actions ── async function removeArtifactTarget(name) { if (!confirm(`Remove artifact target "${name}"?`)) return; try { await api(`/artifacts/targets/${name}`, { method: 'DELETE' }); toast(`Removed target: ${name}`, 'success'); cardInvalidate('artifacts'); loadArtifactsCard(); } catch (e) { toast(`Error: ${e.message}`, 'error'); } } function addArtifactTargetModal() { toast('Add artifact target — coming soon', 'info'); } function buildArtifactTarget(name) { // Delegate to the full build modal system buildArtifactStream(name); } async function publishArtifactTarget(name) { // Check build status first — offer build if not yet built try { const data = await api('/artifacts/targets/' + encodeURIComponent(name) + '/status'); const build = data.build; if (!build || !build.ok) { // No successful build — offer to build first const doBuild = confirm( `"${name}" has no successful build yet.\n\n` + `Build it first before publishing?` ); if (doBuild) { buildArtifactStream(name); } return; } // Has a successful build — go straight to promotion panel _showStandalonePublishPanel(name); } catch (e) { // Fallback — try showing publish panel anyway _showStandalonePublishPanel(name); } } async function _showStandalonePublishPanel(name) { // Create a lightweight modal for publish (not a build modal) const oldModal = document.getElementById('artifact-build-modal'); if (oldModal) oldModal.remove(); const modal = document.createElement('div'); modal.className = 'vault-modal-overlay'; modal.id = 'artifact-build-modal'; modal.onclick = (e) => { if (e.target === modal) closeArtifactBuildModal(); }; modal.innerHTML = `

🚀 Publish: ${esc(name)}

`; document.body.appendChild(modal); const stagesEl = document.getElementById('art-build-stages'); const actionsEl = document.getElementById('art-build-actions'); await _showPromotionPanel(name, actionsEl, stagesEl); } async function buildAllArtifacts() { try { const data = await api('/artifacts/targets'); const targets = data.targets || []; if (targets.length === 0) { toast('No artifact targets configured', 'info'); return; } // Build sequentially — each one gets its own modal let i = 0; function buildNext() { if (i >= targets.length) return; const t = targets[i++]; buildArtifactStream(t.name, { onComplete: (ok) => { if (ok && i < targets.length) { setTimeout(buildNext, 500); } }, }); } buildNext(); } catch (e) { toast(`Error: ${e.message}`, 'error'); } }
